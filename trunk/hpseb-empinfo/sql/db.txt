CREATE TABLE IF NOT EXISTS `m_absemce` (
  `ABSENCE_CODE` varchar(50) NOT NULL COMMENT 'Absence Code with not spl chars and underscore only hyphens allowed',
  `ABSENCE_NAME` varchar(100) NOT NULL COMMENT 'User Friendly Name Given to the absence code',
  `ABSENCE_DESCRIPTION` varchar(500) NOT NULL COMMENT 'Detailed description about the absece code',
  `STATUS` int(1) NOT NULL DEFAULT '0' COMMENT 'Record Status set values for BLOCKED, ACTIVE, INACTIVE, SUSPENDED DELETED etc, use -ve values for negetive status.',
  `CREATED_BY` varchar(50) NOT NULL,
  `CREATED_DATE` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `MODIFIED_BY` varchar(50) NOT NULL,
  `MODIFIED_DATE` timestamp NOT NULL DEFAULT '0000-00-00 00:00:00',
  PRIMARY KEY (`ABSENCE_CODE`),
  UNIQUE KEY `ABSENCE_NAME` (`ABSENCE_NAME`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;
